📋 JavaScript 常用 DOM 操作分类速查

操作类别 方法 / 属性 说明 示例
选取元素 document.getElementById(id) 通过 id 获取单个元素 const el = document.getElementById('box');
document.querySelector(selector) CSS 选择器，返回第一个匹配元素 const el = document.querySelector('.active');
document.querySelectorAll(selector) CSS 选择器，返回所有匹配的 NodeList const list = document.querySelectorAll('li');
document.getElementsByClassName(class) 通过类名获取 HTMLCollection const items = document.getElementsByClassName('item');
document.getElementsByTagName(tag) 通过标签名获取 HTMLCollection const divs = document.getElementsByTagName('div');
element.closest(selector) 查找匹配的最近祖先元素（选取相关） const card = el.closest('.card');
创建元素 document.createElement(tag) 创建新元素节点 const div = document.createElement('div');
document.createTextNode(text) 创建文本节点 const text = document.createTextNode('Hello');
element.cloneNode(deep) 克隆节点，deep 为 true 时包含子节点 const clone = el.cloneNode(true);
添加/插入 parent.appendChild(child) 在父元素末尾添加子节点 parent.appendChild(newEl);
parent.insertBefore(newNode, refNode) 在参考节点前插入新节点 parent.insertBefore(newEl, existingEl);
parent.append(...nodes) 在父元素末尾添加多个节点或文本 parent.append('文本', newEl);
parent.prepend(...nodes) 在父元素开头添加多个节点或文本 parent.prepend(newEl);
element.before(...nodes) 在当前元素之前插入兄弟节点 el.before(newEl, '文本');
element.after(...nodes) 在当前元素之后插入兄弟节点 el.after(newEl);
element.insertAdjacentHTML(pos, html) 在指定位置插入 HTML 字符串 el.insertAdjacentHTML('beforeend', '<p>Hi</p>');
element.insertAdjacentElement(pos, newEl) 在指定位置插入元素节点 el.insertAdjacentElement('afterbegin', newEl);
删除元素 parent.removeChild(child) 从父元素中移除子节点 parent.removeChild(el);
element.remove() 直接从 DOM 中移除自身 el.remove();
element.innerHTML = '' 清空元素所有子内容（HTML） el.innerHTML = '';
element.textContent = '' 清空文本内容 el.textContent = '';
替换元素 parent.replaceChild(newChild, oldChild) 用新节点替换旧的子节点 parent.replaceChild(newEl, oldEl);
element.replaceWith(...nodes) 将自身替换为新节点或文本 el.replaceWith(newEl, '文本');
修改内容 element.textContent 获取/设置纯文本内容 el.textContent = 'Hello';
element.innerText 获取/设置可见文本（受 CSS 影响） el.innerText = 'Hello';
element.innerHTML 获取/设置 HTML 内容 el.innerHTML = '<span>Hi</span>';
属性操作 element.setAttribute(name, value) 设置属性 el.setAttribute('data-id', '1');
element.getAttribute(name) 获取属性值 el.getAttribute('href');
element.removeAttribute(name) 移除属性 el.removeAttribute('disabled');
element.hasAttribute(name) 检查属性是否存在 el.hasAttribute('checked');
element.id / className / href … 直接访问标准属性 el.id = 'main';
element.classList.add(className) 添加类 el.classList.add('active');
element.classList.remove(className) 移除类 el.classList.remove('hidden');
element.classList.toggle(className) 切换类 el.classList.toggle('open');
element.classList.contains(className) 检查是否包含类 if (el.classList.contains('on'))
element.dataset.key 访问 data-* 属性 el.dataset.userId = '123';
样式操作 element.style.propertyName = value 设置行内样式 el.style.backgroundColor = 'red';
element.style.setProperty(prop, value) 设置 CSS 属性（支持自定义属性） el.style.setProperty('--color', 'blue');
getComputedStyle(element) 获取最终计算样式 const styles = getComputedStyle(el);
element.style.cssText 批量设置行内样式 el.style.cssText = 'color: red; font-size: 16px;';
事件处理 element.addEventListener(type, listener) 绑定事件 el.addEventListener('click', handleClick);
element.removeEventListener(type, listener) 移除事件监听 el.removeEventListener('click', handleClick);
事件对象 (e) 常用属性/方法 e.target, e.preventDefault(), e.stopPropagation() e.preventDefault();
{ once: true } 一次性事件 配置项，监听一次后自动移除 el.addEventListener('click', fn, { once: true });
遍历节点 element.parentNode / parentElement 获取父节点 const parent = el.parentElement;
element.children 获取子元素集合（HTMLCollection） const children = el.children;
element.firstElementChild / lastElementChild 第一个/最后一个子元素 const first = el.firstElementChild;
element.nextElementSibling / previousElementSibling 下一个/上一个兄弟元素 const next = el.nextElementSibling;
element.childElementCount 子元素的数量 const count = el.childElementCount;
其他常用 element.matches(selector) 判断元素是否匹配选择器 el.matches('.active');
element.contains(otherElement) 是否包含指定节点 el.contains(childEl);
element.getBoundingClientRect() 获取尺寸和相对视口位置 const rect = el.getBoundingClientRect();
element.scrollIntoView() 滚动元素到可见区域 el.scrollIntoView({ behavior: 'smooth' });
