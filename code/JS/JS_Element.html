分类 方法 / 属性 说明 示例代码
获取元素 getElementById 通过 ID 获取元素 
document.getElementById('id')
querySelector 通过 CSS 选择器获取第一个匹配元素 
document.querySelector('.class')
document.querySelector('#id')
querySelectorAll 通过 CSS 选择器获取所有匹配元素（静态 NodeList） 
document.querySelectorAll('.class')
getElementsByClassName 通过类名获取元素（动态集合） 
document.getElementsByClassName('class')
getElementsByTagName 通过标签名获取元素（动态集合） 
document.getElementsByTagName('div')
getElementsByName 通过 name 属性获取元素（动态集合） 
document.getElementsByName('username')
创建和添加元素 createElement 创建新元素 
const div = document.createElement('div')
createTextNode 创建文本节点 
const text = document.createTextNode('内容')
appendChild 向父元素末尾添加子节点 
parent.appendChild(child)
insertBefore 在指定子节点之前插入新节点 
parent.insertBefore(newNode, referenceNode)
append 在元素末尾追加多个节点或文本 
element.append(node1, '文本', node2)
prepend 在元素开头插入多个节点或文本 
element.prepend(node)
innerHTML 通过 HTML 字符串设置内容 
element.innerHTML = '<span>文本</span>'
insertAdjacentHTML 在指定位置插入 HTML 字符串 
element.insertAdjacentHTML('beforeend', '<span>文本</span>')
// 位置: beforebegin, afterbegin, beforeend, afterend
删除和替换 remove 直接删除元素自身 
element.remove()
removeChild 通过父元素删除子元素 
parent.removeChild(child)
replaceChild 替换子元素（父元素调用） 
parent.replaceChild(newChild, oldChild)
replaceWith 用新节点替换当前元素 
element.replaceWith(newElement)
属性和内容操作 id / className
(直接属性)		 直接设置标准属性 
element.id = 'myId'
element.className = 'box'
getAttribute 读取指定属性值 
element.getAttribute('href')
setAttribute 设置属性值 
element.setAttribute('data-id', '123')
removeAttribute 移除属性 
element.removeAttribute('title')
hasAttribute 判断是否拥有某属性 
element.hasAttribute('disabled')
toggleAttribute 切换属性的存在状态 
element.toggleAttribute('hidden')
classList.add 添加类名 
element.classList.add('active')
classList.remove 移除类名 
element.classList.remove('old')
classList.toggle 切换类名（有则删，无则加） 
element.classList.toggle('dark')
classList.contains 判断是否包含指定类名 
element.classList.contains('active')
textContent 读取或设置纯文本内容 
element.textContent = '纯文本'
innerHTML 读取或设置 HTML 内容 
element.innerHTML = '<b>HTML内容</b>'
outerHTML 读取或替换包含自身的 HTML 
console.log(element.outerHTML)
样式操作 style.属性 设置内联样式 
element.style.color = 'red'
element.style.backgroundColor = '#fff'
style.cssText 一次设置多个内联样式 
element.style.cssText = 'color: red; font-size: 16px;'
getComputedStyle 获取元素最终计算样式 
getComputedStyle(element).color
getComputedStyle(...).getPropertyValue 获取特定 CSS 属性值 
getComputedStyle(element).getPropertyValue('font-size')
遍历节点 parentNode 父节点（所有节点） ← 这些属性所有节点类型（含文本、注释） 
element.parentNode
element.childNodes
element.firstChild
element.lastChild
element.nextSibling
element.previousSibling
childNodes 所有子节点
firstChild 第一个子节点
lastChild 最后一个子节点
nextSibling 下一个兄弟节点
previousSibling 上一个兄弟节点
parentElement 父元素（仅元素节点） ← 这些属性仅元素节点 
element.parentElement
element.children
element.firstElementChild
element.lastElementChild
element.nextElementSibling
element.previousElementSibling
children 所有子元素
firstElementChild 第一个子元素
lastElementChild 最后一个子元素
nextElementSibling 下一个兄弟元素
previousElementSibling 上一个兄弟元素
事件处理 addEventListener 绑定事件监听 
element.addEventListener('click', handler)
// 一次性事件
element.addEventListener('click', handler, { once: true })
removeEventListener 移除事件监听 
element.removeEventListener('click', handler)
事件委托（技术） 在父元素上监听子元素事件 
parent.addEventListener('click', (e) => {
  if (e.target.matches('.child')) {
    // 处理子元素点击
  }
})
dispatchEvent 手动触发事件 
element.dispatchEvent(new Event('click'))
元素位置和尺寸 getBoundingClientRect 获取相对视口的位置及尺寸 
const rect = element.getBoundingClientRect()
// rect.top, rect.left, rect.right, rect.bottom, rect.width, rect.height
scrollTop / scrollLeft 已滚动的像素数 
element.scrollTop
element.scrollLeft
scrollIntoView 滚动元素到可视区域 
element.scrollIntoView({ behavior: 'smooth', block: 'center' })
offsetTop / offsetLeft /
offsetWidth / offsetHeight / offsetParent 		相对定位父元素的偏移与尺寸（含边框） 
element.offsetTop
element.offsetLeft
element.offsetWidth
element.offsetHeight
element.offsetParent
clientWidth / clientHeight 内容+内边距区域尺寸（不含边框） 
element.clientWidth
element.clientHeight
scrollWidth / scrollHeight 可滚动内容的总宽度/高度 
element.scrollWidth
element.scrollHeight
表单操作 value 读取或设置输入值 
input.value = '新值'
checked 复选框/单选框选中状态 
checkbox.checked = true
form.reset() 重置表单 
form.reset()
form.submit() 手动提交表单 
form.submit()
focus() 使元素获得焦点 
input.focus()
blur() 使元素失去焦点 
input.blur()
select() 选中输入框内容 
input.select()
new FormData(form) 创建表单数据对象 
const data = new FormData(form)
实用方法 contains 判断是否包含某节点 
parent.contains(child)
matches 判断元素是否匹配选择器 
element.matches('.selector')
closest 沿 DOM 树向上查找最近的匹配祖先 
element.closest('.parent')
cloneNode 克隆节点（true为深拷贝） 
element.cloneNode(true)
dataset 读取/写入 data-* 属性 
element.dataset.userId  // 对应 data-user-id
requestFullscreen 请求全屏显示 
element.requestFullscreen()
exitFullscreen 退出全屏 
document.exitFullscreen()
