HTML元素 常见监听事件 触发时机 监听示例（JavaScript）
<button> 或 <input type="button/submit/reset"> click 点击按钮时 element.addEventListener('click', handler)
<input type="text/password/email/number"> input 输入框内容实时改变时 element.addEventListener('input', handler)
<input type="text/password/email/number"> change 内容改变并失去焦点后 element.addEventListener('change', handler)
<input type="text/..."> focus / blur 获得 / 失去焦点时 element.addEventListener('focus', handler)
<input type="checkbox"> / <input type="radio"> change 选中状态发生改变时 checkbox.addEventListener('change', handler)
<select> change 选中项发生变化时 select.addEventListener('change', handler)
<textarea> input 文本内容实时改变时 textarea.addEventListener('input', handler)
<textarea> change 内容改变并失去焦点时 textarea.addEventListener('change', handler)
<form> submit 表单提交时（可调用preventDefault阻止提交） form.addEventListener('submit', handler)
<form> reset 表单重置时 form.addEventListener('reset', handler)
<a> click 点击链接时（可阻止默认跳转） link.addEventListener('click', handler)
<img> load 图像加载完成时 img.addEventListener('load', handler)
<img> error 图像加载失败时 img.addEventListener('error', handler)
<input type="file"> change 用户选择文件后 fileInput.addEventListener('change', handler)
<input type="range"> input 滑动值实时改变时 rangeInput.addEventListener('input', handler)
<input type="range"> change 滑动结束释放鼠标时 rangeInput.addEventListener('change', handler)
window load 页面所有资源（图片、样式等）加载完成后 window.addEventListener('load', handler)
window resize 浏览器窗口尺寸改变时 window.addEventListener('resize', handler)
window scroll 窗口滚动时 window.addEventListener('scroll', handler)
document DOMContentLoaded DOM树构建完成，不等待图片等资源 document.addEventListener('DOMContentLoaded', handler)
document keydown / keyup 全局键盘按键按下 / 抬起时 document.addEventListener('keydown', handler)
任意元素（<div>、<span>等） click 鼠标点击元素时 element.addEventListener('click', handler)
任意元素 mouseenter / mouseleave 鼠标指针进入 / 离开元素区域时 element.addEventListener('mouseenter', handler)
可滚动元素（如overflow:auto的div） scroll 元素内部滚动时 scrollableDiv.addEventListener('scroll', handler)
